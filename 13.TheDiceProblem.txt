class Solution 
{ 
public: 
    int oppFace(int N) 
    { 
        if(N==1) 
        return 6; 
        else if(N==2) 
        return 5; 
        else if(N==3) 
        return 4; 
        else if(N==4) 
        return 3; 
        else if(N==5) 
        return 2; 
        else 
        return 1; 
    } 
};
