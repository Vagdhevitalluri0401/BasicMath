class Solution{ 
public:         
        vector<float> streamAvg(int arr[], int n) { 
            float sum=0,count=0; 
            vector<float> v; 
            for(int i=0;i<n;i++) 
            { 
                sum+=arr[i]; 
                count++; 
                v.push_back(sum/count); 
            } 
            return v; 
        } 
};
