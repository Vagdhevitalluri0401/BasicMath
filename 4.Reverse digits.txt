class Solution 
{ 
        public: 
                long long int reverse_digit(long long int n) 
                { 
                    long long int c=n; 
                    long long int rev=0,r=0; 
                    while(n>0) 
                    { 
                        r=n%10; 
                        rev=rev*10+r; 
                        n=n/10; 
                    } 
                    return abs(rev); 
                } 
};
